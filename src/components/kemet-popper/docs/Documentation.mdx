# Kemet Popper

Positions an element over a trigger. This component is a wrapper component for [Popper.js](https://popper.js.org).

    ### Markup
    ```html
    <kemet-popper>
      <kemet-button slot="trigger">[trigger]</kemet-button>
      <div slot="content">[content]</div>
    </kemet-popper>
    ```

    ### Import JavaScript
    ```javascript
    import 'kemet-ui/dist/components/kemet-popper/kemet-popper.js';
    import 'kemet-ui/dist/components/kemet-popper/kemet-popper-close.js'; // optional
    ```

    ### Import HTML
    ```html
    <script src="https://unpkg.com/kemet-ui@latest/dist/components/kemet-popper/kemet-popper.js" type="module"></script>
    <script src="https://unpkg.com/kemet-ui@latest/dist/components/kemet-popper/kemet-popper-close.js" type="module"></script>
    ```
